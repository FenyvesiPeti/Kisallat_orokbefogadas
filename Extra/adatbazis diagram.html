Oszlop Típus Leírás
id INT(11) AUTO_INCREMENT PRIMARY KEY Automatikusan növekszik, elsődleges kulcs
animal_name VARCHAR(100) NOT NULL Állat neve
animal_age INT(11) NOT NULL Kor (években)
animal_type ENUM('kutya', 'macska') NOT NULL Macska vagy kutya
animal_gender ENUM('hím', 'nőstény') NOT NULL Neme (ivar)
animal_size ENUM('kicsi', 'közepes', 'nagy') NOT NULL Méret (pl. kicsi, közepes, nagy)
animal_breed VARCHAR(100) DEFAULT NULL Típus (pl. tacskó, keverék)
animal_description TEXT NOT NULL Leírás
animal_image MEDIUMBLOB NOT NULL Kép
